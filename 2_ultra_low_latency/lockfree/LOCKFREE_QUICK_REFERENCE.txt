│     queue.pop_wait(order);                                               │
│     process_order(order);                                                │
│ }                                                                         │
│                                                                           │

│ ────────────────────────────────────────────────────────────────────────  │

│                                                                           │
│ // Process A (creator)                                                    │
│ int main() {                                                              │
│     SPSCSharedRingBuffer<Order, 4096> queue("/my_queue");                │
│     Order order = create_order();                                        │
│     queue.push_wait(order);  // 100-300ns to other process               │
│ }                                                                         │
│                                                                           │
│ // Process B (attacher)                                                   │
│ int main() {                                                              │
│     auto queue = SPSCSharedRingBuffer<Order, 4096>::attach("/my_queue"); │
│     Order order;                                                          │
│     queue.pop_wait(order);                                               │
│     process_order(order);                                                │
│ }                                                                         │

┌───────────────────────────────────────────────────────────────────────────┐
│ ⚡ BEST PRACTICES                                                         │
├───────────────────────────────────────────────────────────────────────────┤
│ ✅ DO:                                                                    │
│   • Use power-of-2 sizes (1024, 2048, 4096, 8192)                        │
│   • Pin threads/processes to CPU cores                                   │
│   • Use SPSC when possible (fastest)                                     │
│   • Pre-allocate at startup                                              │
│   • Use busy-wait (push_wait/pop_wait) for lowest latency                │
│   • For IPC: Start creator process before attacher                       │
│   • For IPC: Queue names must start with '/' (e.g., "/my_queue")         │
│                                                                           │
│ ❌ DON'T:                                                                 │
│   • Don't use non-power-of-2 sizes (won't compile)                       │
│   • Don't use MPMC when SPSC/MPSC works (slower)                         │
│   • Don't block with sleep() (adds latency)                              │
│   • For IPC: Don't start attacher before creator (will fail)             │
│   • For IPC: Don't use different template params in processes            │
